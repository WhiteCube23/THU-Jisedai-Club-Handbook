\documentclass{ctexart}
\usepackage{tikz}
\usetikzlibrary{shapes.geometric}
\usepackage{xcolor}

% 定义不同样式的平行四边形
\newcommand{\parallelogram}[2][blue!20]{%
    \begin{tikzpicture}[baseline=(text.base)]
        \node[
            parallelogram, 
            fill=#1, 
            draw=#1, 
            text width=3cm, 
            minimum height=1cm, 
            inner sep=4pt
        ] (text) {#2};
    \end{tikzpicture}%
}

\begin{document}

% 示例使用
\parallelogram{这是一个单行平行四边形文本框}

\vspace{1cm}

% 不同颜色和样式
\parallelogram[red!20]{2222}

\parallelogram[green!20]{3333}

\parallelogram[yellow!30]{4444\\112233344}

\end{document}
